class Cachorro:
    def __init__(self,nome,comida,sono):
        self.nome = nome
        self.comida = comida
        self.sono = sono

    def comer(self):
            self.comida -= 1

    def dormir(self):
            self.sono = False

cachorro_1 = Cachorro('Nelson',3,False)

cachorro_1.comer()

from dataclasses import dataclass


@dataclass
class Cliente:
    nome: str
    email: str
    idade: int

    def exibir(self):
        print(f'Meu nome é {self.nome}, tenho {self.idade} anos e meu e-mail é: {self.email}.')

rob = Cliente(nome='Robert', email='robert.melo@gmail.com',idade=29)

rob.exibir()
