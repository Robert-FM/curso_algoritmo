def fibonacci(x):
    t_1 = 0
    print(t_1,end=' ')
    t_2 = 1
    print(t_2,end=' ')

    for t in range(1,x):
        t_3 = t_2 + t_1
        print(t_3,end=' ')
        t_1 = t_2
        t_2 = t_3
